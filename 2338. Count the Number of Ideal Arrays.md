#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int MOD = 1e9 + 7;

    long long modPow(long long base, long long exp, int mod) {
        long long result = 1;
        base = base % mod;
        while (exp > 0) {
            if (exp % 2 == 1)
                result = result * base % mod;
            base = base * base % mod;
            exp /= 2;
        }
        return result;
    }

    long long modInverse(long long a, int m) {
        return modPow(a, m - 2, m);
    }


    int idealArrays(int n, int maxValue) {
        vector<int> minDivisor(maxValue + 1, 0);
        
        for (int p = 2; p <= maxValue; p++) {
            if (minDivisor[p] != 0) 
            continue;

            for (int i = p; i <= maxValue; i += p) {
                if (minDivisor[i] == 0) {
                    minDivisor[i] = p;
                }
            }
        }

        int maxPow = (int)(log(maxValue) / log(2));
        vector<long long> binCoeff(maxPow + 1, 0);
        long long b = 1;
        for (int i = 1; i <= maxPow; i++) {
            b = b * (n + i - 1) % MOD;
            b = b * modInverse(i, MOD) % MOD;
            binCoeff[i] = b;
        }

        int s = 0;

        for (int i = 1; i <= maxValue; i++) {
            int x = i;
            long long product = 1;
            while (x > 1) {
                int p = minDivisor[x];
                int w = 0;
                while (x % p == 0) {
                    w++;
                    x /= p;
                }
                product = product * binCoeff[w] % MOD;
            }
            s = (s + product) % MOD;
        }

        return s;
    }

};
